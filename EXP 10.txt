// Get browser name and version
document.getElementById("browserInfo").innerHTML = "Name: " + navigator.appName + ", Version: " + navigator.appVersion;

// Get screen details
document.getElementById("screenInfo").innerHTML = "Screen Width: " + screen.width + "px, Screen Height: " + screen.height + "px";

// Get history stack information
document.getElementById("historyInfo").innerHTML = "Number of History Entries: " + history.length;

// Count the number of paragraph tags
var paragraphTags = document.getElementsByTagName("p");
document.getElementById("paragraphCount").innerHTML = paragraphTags.length;

// Replace text in a specific tag
document.getElementById("replaceButton").addEventListener("click", function () {
    var tagToReplace = document.getElementById("tagToReplace").value;
    var newText = document.getElementById("newText").value;
    var elementToReplace = document.getElementById(tagToReplace);

    if (elementToReplace) {
        document.getElementById("replacedText").style.display = "block"; // Display the replaced text
        elementToReplace.innerHTML = newText;
        document.getElementById("replacedText").innerHTML = "Replaced Text: " + newText;
    } else {
        alert("Tag not found!");
    }
});
